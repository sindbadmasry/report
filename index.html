const SHEET_NAME = 'Data'; function doGet() { return HtmlService.createHtmlOutputFromFile('Index') .setTitle('نظام إدارة البرامج والبيانات') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // دالة لجلب ورقة العمل أو إنشائها function getSheet() { let ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); // إنشاء العناوين الأساسية sheet.appendRow(['ID', 'Date', 'Program', 'Item', 'Count', 'Price', 'Total']); sheet.getRange('A1:G1').setFontWeight('bold').setBackground('#f3f4f6'); } return sheet; } // دالة التهيئة المبدئية لجلب البيانات function initData() { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const rows = data.slice(1); return rows.map(r => ({ id: r[0].toString(), date: r[1], program: r[2], item: r[3], count: parseFloat(r[4]) || 0, price: parseFloat(r[5]) || 0, total: parseFloat(r[6]) || 0 })); } // دالة إضافة أو تحديث البيانات function saveOrUpdateData(obj) { const sheet = getSheet(); const dateStr = new Date().toISOString().split('T')[0]; // إذا كان هناك ID، فهذا يعني أنه تعديل if (obj.id) { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === obj.id.toString()) { sheet.getRange(i + 1, 3, 1, 5).setValues([[obj.program, obj.item, obj.count, obj.price, obj.total]]); return obj; } } } // إذا لم يكن هناك ID، إضافة صف جديد const newId = new Date().getTime().toString(); sheet.appendRow([newId, dateStr, obj.program, obj.item, obj.count, obj.price, obj.total]); return { id: newId, date: dateStr, ...obj }; } // دالة الحذف function deleteData(id) { const sheet = getSheet(); const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === id.toString()) { sheet.deleteRow(i + 1); return true; } } return false; }
